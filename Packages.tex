% Customise page margin
\usepackage{geometry}
\geometry{
    top = 1.2 in,
    bottom = 1.2 in,
    left = 1.4 in,
    right = 1.4 in
}

% Enable multi-line formula and align function
\usepackage{amsmath}

% Enable mathbb command
\usepackage{amsfonts}

% Enable extension of mathematical symbols
\usepackage{mathtools}

% Enable interactive reference, this is necessary for interactive navigations such as table of contents
\usepackage{hyperref}
\hypersetup{
    colorlinks = true, % Set true if you want colored links
    linkcolor = black,  % Choose some color if you want links to stand out
    linktoc = all,     % Set to all if you want both sections and subsections linked
}

% Enable advanced graphic controls
\usepackage{graphicx}

% Enable floating objects control, this includes the [H] command fixing images' locations
\usepackage{float}

% Plot diagrams
\usepackage{tikz}

% Enable advanced colour control
\usepackage{xcolor}
\definecolor{linecol}{rgb}{0.85, 0.85, 0.85} % Customise colour used for the code blocks' frame lines
\definecolor{codebg}{rgb}{0.97, 0.97, 0.97} % Customise colour used as code blocks' background

% Render code blocks better with pygmentised highlighting
\usepackage{minted}
\newminted[Rminted]{R}{ % Customised minted function for R code blocks
    frame = leftline,
    framesep = 8 pt,
    rulecolor = linecol,
    xleftmargin = 18 pt,
    xrightmargin = 10 pt,
    mathescape, % Enable LaTeX math rendered as math symbols in comments
    obeytabs,
    tabsize = 4,
    bgcolor = codebg,
    linenos, % Show line number
    numbersep = 5 pt,
    breaklines, % Set auto wrap
    breaksymbolleft = , % Remove wrap symbol
}

% Render psudocodes
\usepackage{algpseudocode}
